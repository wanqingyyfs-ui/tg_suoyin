**新增脚本**: `scripts/stats.py`

```python
#!/usr/bin/env python3
import sqlite3
from datetime import datetime, timedelta

def get_stats():
    conn = sqlite3.connect('data/rectg.db')
    cur = conn.cursor()
    
    # 总数统计
    total = cur.execute('SELECT COUNT(*) FROM entries').fetchone()[0]
    valid = cur.execute('SELECT COUNT(*) FROM entries WHERE keep=1 AND valid=1 AND private=0').fetchone()[0]
    filtered = cur.execute('SELECT COUNT(*) FROM entries WHERE keep=0').fetchone()[0]
    private = cur.execute('SELECT COUNT(*) FROM entries WHERE private=1').fetchone()[0]
    
    # 分类统计
    categories = cur.execute('''
        SELECT category, COUNT(*) 
        FROM entries 
        WHERE keep=1 AND valid=1 AND private=0 
        GROUP BY category 
        ORDER BY COUNT(*) DESC
    ''').fetchall()
    
    # 过滤原因统计
    reasons = cur.execute('''
        SELECT filter_reason, COUNT(*) 
        FROM entries 
        WHERE keep=0 AND filter_reason IS NOT NULL 
        GROUP BY filter_reason 
        ORDER BY COUNT(*) DESC
    ''').fetchall()
    
    # 最近添加
    recent = cur.execute('''
        SELECT username, title, created_at 
        FROM entries 
        ORDER BY created_at DESC 
        LIMIT 10
    ''').fetchall()
    
    conn.close()
    
    print(f"📊 数据统计报告 ({datetime.now().strftime('%Y-%m-%d %H:%M:%S')})")
    print(f"\n总览:")
    print(f"  总数: {total}")
    print(f"  有效展示: {valid} ({valid/total*100:.1f}%)")
    print(f"  已过滤: {filtered} ({filtered/total*100:.1f}%)")
    print(f"  私密: {private} ({private/total*100:.1f}%)")
    
    print(f"\n分类分布 (Top 10):")
    for cat, count in categories[:10]:
        print(f"  {cat or '未分类'}: {count}")
    
    print(f"\n过滤原因 (Top 10):")
    for reason, count in reasons[:10]:
        print(f"  {reason}: {count}")
    
    print(f"\n最近添加:")
    for username, title, created_at in recent:
        print(f"  {username} - {title} ({created_at})")

if __name__ == '__main__':
    get_stats()
```

```javascript
function lazyLoadImages() {
  const images = document.querySelectorAll('img[data-src]');
  const observer = new IntersectionObserver(entries => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        const img = entry.target;
        img.src = img.dataset.src;
        img.removeAttribute('data-src');
        observer.unobserve(img);
      }
    });
  });
  images.forEach(img => observer.observe(img));
}
```

创建 `scripts/optimize_db.py`:

```python
#!/usr/bin/env python3
import sqlite3

def optimize_db():
    conn = sqlite3.connect('data/rectg.db')
    cur = conn.cursor()
    
    # 创建索引
    indexes = [
        'CREATE INDEX IF NOT EXISTS idx_entries_username ON entries(username)',
        'CREATE INDEX IF NOT EXISTS idx_entries_category ON entries(category)',
        'CREATE INDEX IF NOT EXISTS idx_entries_keep_valid_private ON entries(keep, valid, private)',
        'CREATE INDEX IF NOT EXISTS idx_entries_type ON entries(type)',
    ]
    
    for idx in indexes:
        cur.execute(idx)
    
    # 分析表
    cur.execute('ANALYZE')
    
    # 清理
    cur.execute('VACUUM')
    
    conn.commit()
    conn.close()
    print('✅ Database optimized')
```

```powershell
$BackupDir = "D:\编程\wanqing\projects\rectg\backups"

$BackupFile = "$BackupDir\rectg_$Timestamp.db"

# 创建备份目录
if (-not (Test-Path $BackupDir)) {
    New-Item -ItemType Directory -Path $BackupDir
}

# 复制数据库
Copy-Item "data\rectg.db" $BackupFile

# 删除 30 天前的备份
Get-ChildItem $BackupDir -Filter "rectg_*.db" |
```
